\textbf{Model Algorithms:}

\begin{enumerate}
    \item \textbf{Multinomial Naive Bayes:}
    \begin{itemize}
        \item Based on Bayes' theorem: $P(\text{Spam|Features}) = \frac{P(\text{Features|Spam}) \times P(\text{Spam})}{P(\text{Features})}$
        \item Assumes feature independence (naive assumption)
        \item Efficient and works well with text classification
    \end{itemize}
    
    \item \textbf{Logistic Regression:}
    \begin{itemize}
        \item Linear classifier using logistic function
        \item Probability-based predictions
        \item Easy to interpret and efficient
    \end{itemize}
    
    \item \textbf{Support Vector Machine (SVM):}
    \begin{itemize}
        \item Finds optimal hyperplane to maximize margin between classes
        \item LinearSVC for linear separation
        \item Powerful for high-dimensional feature spaces
    \end{itemize}
\end{enumerate}

\textbf{Evaluation Metrics:}
\begin{itemize}
    \item \textbf{Accuracy:} $\frac{TP + TN}{TP + TN + FP + FN}$ --- Overall correctness
    \item \textbf{Precision:} $\frac{TP}{TP + FP}$ --- Among spam predictions, how many were correct
    \item \textbf{Recall:} $\frac{TP}{TP + FN}$ --- Among actual spam, how many were detected
    \item \textbf{F1-Score:} $2 \times \frac{\text{Precision} \times \text{Recall}}{\text{Precision} + \text{Recall}}$ --- Harmonic mean
\end{itemize}

\subsection{3. Decision Tree Classifier - From Scratch}

\begin{lstlisting}
class Node:
    """A node in the decision tree"""
    def __init__(self, feature=None, threshold=None, 
                 left=None, right=None, value=None):
        self.feature = feature
        self.threshold = threshold
        self.left = left
        self.right = right
        self.value = value

class DecisionTree:
    """Decision Tree Classifier - from scratch"""
    
    def __init__(self, max_depth=None, min_samples_split=2):
        self.max_depth = max_depth
        self.min_samples_split = min_samples_split
        self.tree = None
    
    def _entropy(self, y):
        """Calculate entropy of labels"""
        counts = np.bincount(y)
        probabilities = counts / len(y)
        entropy = -np.sum([p * np.log2(p) 
                          for p in probabilities if p > 0])
        return entropy
    
    def _information_gain(self, parent, left_child, right_child):
        """Calculate information gain from a split"""
        n = len(parent)
        n_left = len(left_child)
        n_right = len(right_child)
        
        if n_left == 0 or n_right == 0:
            return 0
        
        parent_entropy = self._entropy(parent)
        left_entropy = self._entropy(left_child)
        right_entropy = self._entropy(right_child)
        
        weighted_child_entropy = ((n_left / n) * left_entropy + 
                                 (n_right / n) * right_entropy)
        ig = parent_entropy - weighted_child_entropy
        return ig
    
    def _best_split(self, X, y):
        """Find the best feature and threshold to split on"""
        best_gain = -1
        best_feature = None
        best_threshold = None
        
        for feature_idx in range(X.shape[1]):
            feature_values = X[:, feature_idx]
            thresholds = np.unique(feature_values)
            
            for threshold in thresholds:
                left_mask = feature_values <= threshold
                right_mask = ~left_mask
                
                if (np.sum(left_mask) < self.min_samples_split or 
                    np.sum(right_mask) < self.min_samples_split):
                    continue
                
                ig = self._information_gain(y, y[left_mask], y[right_mask])
                
                if ig > best_gain:
                    best_gain = ig
                    best_feature = feature_idx
                    best_threshold = threshold
        
        return best_feature, best_threshold
    
    def fit(self, X, y):
        """Build decision tree classifier"""
        self.tree = self._build_tree(X, y)
        return self
    
    def predict(self, X):
        """Predict class labels for samples"""
        return np.array([self._predict_sample(x, self.tree) 
                        for x in X])
\end{lstlisting}

\textbf{Decision Tree Algorithm:}
\begin{itemize}
    \item \textbf{Entropy:} Measures impurity of a node
    $$H(S) = -\sum p_i \log_2(p_i)$$
    
    \item \textbf{Information Gain:} Quality of a split
    $$IG(S, A) = H(S) - \sum \frac{|S_v|}{|S|} H(S_v)$$
    
    \item \textbf{Best Split Selection:} Try all features and thresholds
    \item \textbf{Recursive Building:} Build left and right subtrees
    \item \textbf{Stopping Criteria:} Max depth, min samples, or pure nodes
\end{itemize}

\subsection{4. Random Forest Classifier - From Scratch}

\begin{lstlisting}
class RandomForest:
    """Random Forest Classifier - from scratch"""
    
    def __init__(self, n_trees=10, max_depth=None, 
                 min_samples_split=2, bootstrap=True):
        self.n_trees = n_trees
        self.max_depth = max_depth
        self.min_samples_split = min_samples_split
        self.bootstrap = bootstrap
        self.trees = []
    
    def _bootstrap_sample(self, X, y):
        """Create a bootstrap sample of the data"""
        n_samples = X.shape[0]
        indices = np.random.choice(n_samples, n_samples, replace=True)
        return X[indices], y[indices]
    
    def fit(self, X, y):
        """Build random forest"""
        self.trees = []
        for i in range(self.n_trees):
            if self.bootstrap:
                X_sample, y_sample = self._bootstrap_sample(X, y)
            else:
                X_sample, y_sample = X, y
            
            dt = DecisionTree(max_depth=self.max_depth, 
                            min_samples_split=self.min_samples_split)
            dt.fit(X_sample, y_sample)
            self.trees.append(dt)
        
        return self
    
    def predict(self, X):
        """Predict by majority voting"""
        predictions = np.array([tree.predict(X) for tree in self.trees])
        final_predictions = np.zeros(X.shape[0], dtype=int)
        for i in range(X.shape[0]):
            votes = predictions[:, i]
            final_predictions[i] = np.bincount(votes).argmax()
        
        return final_predictions
\end{lstlisting}

\textbf{Random Forest Algorithm:}
\begin{itemize}
    \item \textbf{Bootstrap Sampling:} Create random samples with replacement
    $$S_i = \text{Bootstrap}(S) \text{ for } i = 1 \text{ to } n\_trees$$
    
    \item \textbf{Tree Training:} Train independent decision trees
    \item \textbf{Ensemble Prediction:} Majority voting
    $$\text{Prediction} = \text{argmax}(\text{votes})$$
\end{itemize}

\chapter{Key Mathematical Formulas}

\section{Entropy}

$$H(S) = -\sum_{i=1}^{n} p_i \log_2(p_i)$$

where $p_i$ is the probability of class $i$.

\section{Information Gain}

$$IG(S, A) = H(S) - \sum_{v \in \text{Values}(A)} \frac{|S_v|}{|S|} H(S_v)$$

\section{Accuracy}

$$\text{Accuracy} = \frac{TP + TN}{TP + TN + FP + FN}$$

\section{Precision}

$$\text{Precision} = \frac{TP}{TP + FP}$$

\section{Recall}

$$\text{Recall} = \frac{TP}{TP + FN}$$

\section{F1-Score}

$$F1 = 2 \times \frac{\text{Precision} \times \text{Recall}}{\text{Precision} + \text{Recall}}$$
